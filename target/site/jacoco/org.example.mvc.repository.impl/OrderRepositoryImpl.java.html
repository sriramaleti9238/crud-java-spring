    List<Order> orderList = new ArrayList<Order>( );

        orderList.add(orderData);
        return orderData;
    }


    public Order findById(Long id) {
        for(Order obj : orderList) {
            if(obj.getId().equals(id)) {
                return obj;
            }
        }
        return null;
    }

    @Override
    public void delete(Long id) {
        for(Order obj: orderList) {
            if(obj.getId().equals(id)) {
                orderList.remove(obj);
                break;
            }
        }
    }
}
